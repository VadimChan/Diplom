История C++

C++ был создан Бьёрном Страуструпом в 1979 году в Bell Labs как расширение языка C. Первый стандарт вышел в 1998 году.

Особенности

  • Компилируемый язык
  • Поддержка ООП
  • Низкоуровневый доступ
  • Высокая производительность

Применение

C++ широко используется в разработке игр, операционных систем, высоконагруженных серверов и научных вычислениях.

Основы C++

Фундаментальные концепции и базовый синтаксис языка программирования C++.

Первая программа

Базовая структура программы на C++ включает директивы препроцессора, функцию main() и операторы.

#include <iostream>  // Директива для ввода/вывода

// Главная функция программы
int main() {
    // Вывод сообщения в консоль
    std::cout << "Моя первая программа на C++!\n";

    // Возврат значения
    return 0;
}

Переменные и типы данных

C++ поддерживает различные типы данных:

  • int - целые числа
  • float/double - числа с плавающей точкой
  • char - символы
  • bool - логические значения
int age = 25;
double price = 99.95;
char grade = 'A';
bool is_valid = true;

Операторы

C++ поддерживает различные операторы:

  • Арифметические: +, -, *, /, %
  • Сравнения: ==, !=, >, <, >=, <=
  • Логические: &&, ||, !
  • Присваивания: =, +=, -=, *=, /=
int a = 10, b = 3;
int sum = a + b;  // 13
int mod = a % b;  // 1
bool result = (a > b);  // true

Функции

Функции позволяют организовать код в логические блоки.

#include <iostream>

// Объявление функции
int addNumbers(int a, int b);

int main() {
    int result = addNumbers(5, 7);
    std::cout << "Сумма: " << result;
    return 0;
}

// Определение функции
int addNumbers(int a, int b) {
    return a + b;
}

Контроль выполнения

Управляющие конструкции для контроля потока выполнения программы.

Условные операторы

Оператор if-else позволяет выполнять код по условию.

int number = 10;

if (number > 0) {
    std::cout << "Число положительное";
} else if (number < 0) {
    std::cout << "Число отрицательное";
} else {
    std::cout << "Число равно нулю";
}

Циклы

Циклы позволяют повторять выполнение блока кода.

Цикл for:

for (int i = 0; i < 5; i++) {
    std::cout << i << " ";
}
// Вывод: 0 1 2 3 4

Циклы while

Цикл while:

int i = 0;
while (i < 5) {
    std::cout << i << " ";
    i++;
}
// Вывод: 0 1 2 3 4

Switch-case

Конструкция switch-case для множественного выбора.

char grade = 'B';

switch (grade) {
    case 'A':
        std::cout << "Отлично!";
        break;
    case 'B':
        std::cout << "Хорошо!";
        break;
    case 'C':
        std::cout << "Удовлетворительно";
        break;
    default:
        std::cout << "Неизвестная оценка";
}

Объектно-Ориентированное Программирование

Принципы ООП: инкапсуляция, наследование, полиморфизм.

Классы и объекты

Класс - это пользовательский тип данных, содержащий данные и методы.

class Rectangle {
private:
    double width, height;

public:
    // Конструктор
    Rectangle(double w, double h) : width(w), height(h) {}

    // Метод класса
    double area() {
        return width * height;
    }
};

int main() {
    Rectangle rect(5.0, 3.0);
    std::cout << "Площадь: " << rect.area();
    return 0;
}

Наследование

Наследование позволяет создавать новые классы на основе существующих.

class Animal {
public:
    void eat() {
        std::cout << "Я ем";
    }
};

class Dog : public Animal {
public:
    void bark() {
        std::cout << "Гав!";
    }
};

int main() {
    Dog myDog;
    myDog.eat();  // Унаследованный метод
    myDog.bark();
    return 0;
}

Полиморфизм

Полиморфизм позволяет объектам разных классов обрабатываться одинаково.

class Shape {
public:
    virtual void draw() = 0;  // Чистая виртуальная функция
};

class Circle : public Shape {
public:
    void draw() override {
        std::cout << "Рисую круг";
    }
};

class Square : public Shape {
public:
    void draw() override {
        std::cout << "Рисую квадрат";
    }
};

void drawShape(Shape* shape) {
    shape->draw();
}

Pro уровень

Продвинутые техники программирования на C++.

Шаблоны

Шаблоны позволяют создавать обобщённые функции и классы.

template <typename T>
T max(T a, T b) {
    return (a > b) ? a : b;
}

int main() {
    std::cout << max(5, 10) << "\n";      // 10
    std::cout << max(3.14, 2.71) << "\n"; // 3.14
    return 0;
}

Многопоточность

Библиотека <thread> позволяет создавать многопоточные приложения.

#include <iostream>
#include <thread>

void printMessage(const std::string& message) {
    std::cout << message << "\n";
}

int main() {
    std::thread t1(printMessage, "Поток 1");
    std::thread t2(printMessage, "Поток 2");

    t1.join();
    t2.join();

    return 0;
}

Умные указатели

Умные указатели автоматически управляют памятью.

#include <memory>

int main() {
    // unique_ptr - эксклюзивное владение
    std::unique_ptr<int> ptr1(new int(10));

    // shared_ptr - разделяемое владение
    std::shared_ptr<int> ptr2 = std::make_shared<int>(20);

    return 0;
}

STL (Standard Template Library)

STL предоставляет набор шаблонных классов и функций.

#include <vector>
#include <algorithm>

int main() {
    std::vector<int> numbers = {5, 2, 8, 1, 9};

    // Сортировка
    std::sort(numbers.begin(), numbers.end());

    // Поиск
    auto it = std::find(numbers.begin(), numbers.end(), 8);

    return 0;
}
